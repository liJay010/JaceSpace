//541. 反转字符串 II
class Solution {
public:
    string reverseStr(string s, int k) {

        for (int i = 0; i <= s.size(); i+=(2*k)) {
            if (i + k <= s.size())
            reverse(s.begin()+i,s.begin()+i+k);
            else reverse(s.begin()+i,s.end());
        }
        return s;
    }
    
};

//剑指 Offer 05. 替换空格（双指针法）
class Solution {
public:
    string replaceSpace(string s) {
        int slow ,fast = s.size() -1, count = 0;
        for (char i:s) {
            if (i == ' ')count++;
        }
        s.resize(s.size() + count * 2);
        slow = s.size() -1;
        while (fast >= 0)
        {
            if (s[fast] != ' ')
            {
                s[slow--] = s[fast];

            } else
            {
                s[slow--] = '0';
                s[slow--] = '2';
                s[slow--] = '%';
            }
            fast--;
        }
        return s;
    }
};
```

//151. 反转字符串中的单词

class Solution {
public:
    string reverseWords(string s) {

        /*
         *
         * 1.移除字符串多余空格
         * 2.反转字符串
         * 3.对字符串每个单词进行反转
         */

        //1.移除字符串多余空格 双指针
        int slow = 0,fast = 0;
        while (fast < s.size())
        {
            //cout << s << endl;
            if (s[fast] != ' ')  s[slow++] = s[fast];
            else
            {
                while (s[fast]==' ') fast++;
                fast--;
                if (slow!=0) s[slow++] = ' ';
            }
            fast++;
        }
        if (s[s.size()-1] == ' ') s.resize(slow-1);
        else s.resize(slow);


        //2.反转字符串
        reverse(s.begin(), s.end());


        //3.对字符串每个单词进行反转
        slow = 0;
        for (int i = 0; i <= s.size(); ++i) {
            if (s[i] == ' ' || (i == s.size()))
            {
                reverse(s.begin() + slow, s.begin() + i);
                slow = i+1;
            }
        }
        return s;
    }
};
```

 class Solution {
public:
    string reverseLeftWords(string s, int n) {
        reverse(s.begin(),s.begin()+n);
        reverse(s.begin()+n,s.end());
        reverse(s.begin(),s.end());
        return s;
    }
};

//459. 重复的子字符串
class Solution {
public:
    /*
    技巧：两个相同字符串拼接去掉首尾元素，若能找到原字符串s则为循环字符串
    */
    bool repeatedSubstringPattern(string s) {
        string tmp = s;
        s.append(s);
        s.erase(s.size() -1 );
        s.erase(s.begin());
        int fid = s.find(tmp);
        if (fid!=-1) return true;
        else return false;
    }
};
