面试题15：二进制中1的个数
https://leetcode.cn/problems/er-jin-zhi-zhong-1de-ge-shu-lcof/

class Solution {
    public int hammingWeight(int n) {
        int count = 0;
        for (int i = 0; i < 32; i++) {
            if ((n & (1 << i)) != 0) {
                count++;
            }
        }
        return count;
    }
}
